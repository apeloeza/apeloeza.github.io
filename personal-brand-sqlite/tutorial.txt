js
// server.js
const sqlite3 = require('sqlite3').verbose();
const path = require('path');

// Buka database (akan dibuat jika belum ada)
const dbPath = path.join(__dirname, 'branding.db');
const db = new sqlite3.Database(dbPath, (err) => {
  if (err) {
    console.error('Error opening database:', err.message);
  } else {
    console.log('Connected to SQLite database');
  }
});

// Buat tabel jika belum ada
db.run(`
  CREATE TABLE IF NOT EXISTS branding (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    title TEXT NOT NULL,
    description TEXT NOT NULL,
    profileImage TEXT NOT NULL
  )
`, (err) => {
  if (err) {
    console.error('Error creating table:', err.message);
  } else {
    console.log('Table created or already exists');
  }
});

// Tambahkan data default jika belum ada
db.get("SELECT * FROM branding LIMIT 1", (err, row) => {
  if (err) {
    console.error(err.message);
  } else if (!row) {
    // Insert data default
    db.run(`
      INSERT INTO branding (name, title, description, profileImage)
      VALUES (?, ?, ?, ?)
    `, [
      "SANTUSH DEB",
      "Creative Designer & Developer",
      "I help brands grow with impactful digital experiences.",
      "https://via.placeholder.com/200"
    ], (err) => {
      if (err) {
        console.error("Error inserting default data:", err.message);
      } else {
        console.log("Default data inserted");
      }
    });
  }
});

// API untuk mengambil data
const express = require('express');
const app = express();
const cors = require('cors');

app.use(cors());
app.use(express.json());

// Endpoint untuk mendapatkan data
app.get('/api/personal-brand', (req, res) => {
  db.get("SELECT * FROM branding WHERE id = 1", (err, row) => {
    if (err) {
      res.status(500).json({ error: err.message });
    } else {
      res.json(row);
    }
  });
});

// Endpoint untuk update data
app.put('/api/personal-brand', (req, res) => {
  const { name, title, description, profileImage } = req.body;
  db.run(`
    UPDATE branding SET
      name = ?,
      title = ?,
      description = ?,
      profileImage = ?
    WHERE id = 1
  `, [name, title, description, profileImage], function(err) {
    if (err) {
      res.status(500).json({ error: err.message });
    } else {
      res.json({ message: "Data updated", changes: this.changes });
    }
  });
});

// Start server
const PORT = 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
